1 Functional Programming

1.1 Higher-order Functions

Before: 

function triple(x) {
    return x * 3
}

After: 

var triple = function(x) {
    return x * 3
}

使用高阶函数过滤出指定数组 ➡️ Before: 

let aaa = []
for (let i = 0; i < shuzu.length; i++) {
    if(shuzu[i].item === 'xxx')
        aaa.push(shuzu[i])
}

After: 

let aaa = shuzu.filter(function(xxx) {
    return xxx.species === 'item'
})

分离: 

let isXx = function(xx) {
    return xx.item === 'xx'
} //确定xx为xx

let x = xx.filter(isXx) //用结果创建数组

1.2 map

和filter不同的是,map包含数组所有项目 Before: 

let names = []
for (var i = 0; i < shuzu.length; i ++) {
    names.push(shuzu[i].name)
}

After: 

let names = shuzu.map(function(xx) {
    return xx.name + '可以返回任何对象' + xx.xx
})

➡️化: 

let names = shuzu.map((xx) => xx.name )

1.3 reduce:

Before: 

var xxxs = {
    { item: 111 },
    { item: 111 },
    { item: 111 },
    { item: 111 },
    { item: 111 },
    { item: 111 },
    { item: 111 },
}

let total = 0
for (let i = 0; i < xxxs.length; i++) {
    total += xxxs[i].item
}

After: 

let total = xxxs.deduce(function(sum, xxxs) {
    return sum + xxxs.item
}, 0)
//从该数组第0号开始依次执行
